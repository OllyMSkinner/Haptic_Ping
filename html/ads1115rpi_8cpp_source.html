143void ADS1115rpi::worker()
144{
145 static constexpr int WARN_EVERY = 10;
146 int misses = 0;
147
148 while (running)
149 {
150 bool edgeSeen = request->wait_edge_events(

152
153 if (!running) break;
154

228unsigned ADS1115rpi::i2c_readWord(uint8_t reg)
229{
230 std::lock_guard<std::mutex> lk(i2c1_mutex());

232 i2c_selectDevice();

234 uint8_t tmp[2];

250int ADS1115rpi::i2c_readConversion()
251{

253

255

260 msgs[0].addr = ads1115settings.address;
261 msgs[0].flags = 0;
262 msgs[0].len = 1;
263 msgs[0].buf = &reg;
264
265 msgs[1].addr = ads1115settings.address;
266 msgs[1].flags = I2C_M_RD;
267 msgs[1].len = 2;
268 msgs[1].buf = buf;
269
270 struct i2c_rdwr_ioctl_data txn;
271 txn.msgs = msgs;
272 txn.nmsgs = 2;
273
274 for (int attempt = 0; attempt < MAX_RETRIES; ++attempt) {
275 int ret = ioctl(fd_i2c, I2C_RDWR, &txn);
276 if (ret >= 0) {
277 return (static_cast<int>(buf[0]) << 8)
278 | static_cast<int>(buf[1]);
279 }
280 }
