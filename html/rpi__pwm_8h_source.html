157 void loop() {
159 const int period_us = 1000000 / PWM_FREQ_HZ;
160
162 int tfd = timerfd_create(CLOCK_MONOTONIC, 0);
163 if (tfd < 0) return;
164
166 while (running_) {
167 // Read the current duty cycle.
168 float d = duty_.load();
169
171 int on_us = (int)(period_us * d / 100.0f);
172 int off_us = period_us - on_us;
173
175 if (d >= 100.0f) {
176 gpiod_line_request_set_value(request_, PWM_GPIO_LINE, GPIOD_LINE_VALUE_ACTIVE);
177 waitUs(tfd, period_us);
178
180 } else if (d <= 0.0f) {
181 gpiod_line_request_set_value(request_, PWM_GPIO_LINE, GPIOD_LINE_VALUE_INACTIVE);
182 waitUs(tfd, period_us);
183
185 } else {
186 gpiod_line_request_set_value(request_, PWM_GPIO_LINE, GPIOD_LINE_VALUE_ACTIVE);
187 waitUs(tfd, on_us);
188 gpiod_line_request_set_value(request_, PWM_GPIO_LINE, GPIOD_LINE_VALUE_INACTIVE);
189 waitUs(tfd, off_us);
